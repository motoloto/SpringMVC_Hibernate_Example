CREATE SEQUENCE employee_seq
INCREMENT BY 1
START WITH 1
NOMAXVALUE
NOCYCLE
NOCACHE;

查詢目前表格中有使用中sequence
 SELECT * FROM USER_SEQUENCES;

刪除sequence:
 DROP SEQUENCE employee_seq;
 
●取用一組新的值:
 SELECT employee_seq.NEXTVAL FROM DUAL;

得知目前使用到幾號 :
 SELECT employee_seq.CURRVAL FROM DUAL;
